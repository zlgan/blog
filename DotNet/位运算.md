using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Text;
using Shouldly;

namespace ConsoleApp1
{
    public class BitOperate
    {
        public static void Test()
        {
            var a = 0b1001; //二级制0b开头,a默认是int32类型，内存中值实际是0x00000009
            var b = 0x20; //十六进制0x开头
            var c = Convert.ToInt32("1100", 2);//12，字符串转成二进制
            var d = Convert.ToInt32("1100", 16); //4352 字符串转成十六进制
            var e = 0xfffffff6;
            var f = e == 0xfffffff6;

            var a1 = a & 0b101; //与
            Debug.Assert(a1 == 0b1);
            var a2 = a | 0b111; //或
            Debug.Assert(a2 == 0b1111);
            Int64 a3 = ~a; //取反
            Debug.Assert(a3 == 0xfffffff6);
            var a4 = a ^ 0b101; //异或
            Debug.Assert(a4 == 0b1100);

            var a5 = a >> 2; //右移两位相当于乘以2的n次方
            Debug.Assert(a5 == 0b10);

            var a6 = a << 2;
            Debug.Assert(a6 == 0b100100); //左移两位相当于除以2的n次方

        }


        public static void Test2()
        {
            ButtonStyle myEnum = ButtonStyle.ok | ButtonStyle.submit | ButtonStyle.edit | ButtonStyle.test3;

            foreach (var item in Enum.GetValues(typeof(ButtonStyle)))
            {
                if (((int)myEnum & (int)item) == (int)item)
                {
                    Console.WriteLine(item.ToString());
                }
            }
            Console.WriteLine("");
            myEnum ^= ButtonStyle.test3;//异或运算，从组合中删除指定的项目
            foreach (var item in Enum.GetValues(typeof(ButtonStyle)))
            {
                if (((int)myEnum & (int)item) == (int)item)
                {
                    Console.WriteLine(item.ToString());
                }
            }

        }
        enum ButtonStyle
        {
            cancel = 0x01,
            edit = 0x02,
            delete = 0x04,
            submit = 0x08,
            test1 = 0x10,
            test2 = 0x20,
            test3 = 0x40,
            ok = 0x80
        }
		//简单写法
        enum ButtonStyle2
        {
            cancel = 0x01,
            edit = 0x01 << 1,

            test1 = 0x01 << 4,
            test2 = 0x01 << 5,
            test3 = 0x01 << 6,
            ok = 0x01 << 7
        }
    }
